# Dependency
Some libraries have dependencies to others:
- nasm
- pkg-config
- zlib (for ffmpeg output of png format)
- openssl
- cmake
    - requires openssl
- libxml2
- fribidi
- freetype
- fontconfig
    - requires libxml2
- harfbuzz
- sdl (for ffplay)
- libbluray
    - requires libxml2
    - requires freetype
    - requires fontconfig
- libass
    - requires fribidi
    - requires freetype
    - requires fontconfig
    - requires harfbuzz
- libogg
- zvbi
- aom
- opensh264
- svt-av1
- vpx
- x264
- x265
- lame
- opus
- libvorbis
    - requires libogg
- libtheora
    - requires libogg
    - requires libvorbis
